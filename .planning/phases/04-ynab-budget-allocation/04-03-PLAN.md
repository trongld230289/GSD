computeBudgetRows logic (implement as useMemo inside BudgetPage):
```typescript
function computeBudgetRows(budgets, transactions, categories): BudgetRow[] {
  return categories
    .filter(c => c.type === 'expense')
    .map(cat => {
      const budgeted = budgets.find(b => b.category_id === cat.id)?.budgeted ?? 0
      const spent = transactions
        .filter(t => t.type === 'expense' && t.category_id === cat.id)
        .reduce((sum, t) => sum + t.amount, 0)
      return { category: cat, budgeted, spent, available: budgeted - spent }
    })
}
```
